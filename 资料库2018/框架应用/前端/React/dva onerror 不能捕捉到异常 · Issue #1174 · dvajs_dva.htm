reducers.js
export function deleteEffect(defaultAction, actionToExecute) {
function *effect(action, param) {
let mergedAction = action;
if (defaultAction) {
mergedAction = { ...defaultAction, ...action };
}
const { call, put } = param;
const { resolve, reject } = mergedAction;
try {
const result = yield call(actionToExecute, mergedAction, param);
yield put({ ...mergedAction, type: 'deleteS', ...result });
if (resolve) {
resolve();
}
} catch (e) {
if (reject) {
reject();
}
throw e;
}
}
return effect;
}

service/userService.js
export function deleteUser(userAccount, options) {
return request(/users/account/${userAccount}, {
method: 'DELETE',
...options,
});
}

request.js
export default async function request(url, options) {
const addr = url.startsWith('http') ? url : ${SERVICE_ROOT}${url};
const defOptions = defaultOptions();
const headers = { ...defOptions.headers, ...(options ? options.headers : {}) };
const ops = { ...defOptions, ...options };
ops.headers = headers;

  let response;
  let responseText;
  try {
	response = await fetch(addr, ops);
	responseText = await response.text();
	if (response.status >= 200 && response.status < 300) {
	  const data = (responseText === undefined || responseText === '') ? {} : 
              JSON.parse(responseText);
	  return { data };
	} else {
	  const errors = ops.errors;
	  if (errors) {
		const error = {
		  status: response.status,
		  respDesc: response.statusText,
		  ...parseErrorResp(responseText),
		};
		if (errors.indexOf(error.respCode) >= 0 || errors.indexOf(response.status) >= 0) {
		  return { error };
		}
	  }
	  throw new Error(`unhandled error ${response.status}`);
	}
  } catch (e) {
	let error = { respDesc: e.message };
	let errorMsg;
	if (response) {
	  let respDesc = response.statusText;
	  if (response.status === 401) respDesc = '操作需要登录才能访问';
	  else if (response.status === 403) respDesc = '操作未授权';
	  error = {
		status: response.status,
		respDesc,
		...parseErrorResp(responseText),
	  };
	  if (error.respCode && error.respCode.endsWith(ERROR_VALIDATION)) {
		errorMsg = '请求参数校验异常';
	  }
	}
	if (ops.errors && ops.errors.indexOf('all') >= 0) {
	  return { error };
	}
	try {
	  const ex = new Error(errorMsg || error.respDesc);
	  ex.response = response;
	  ex.error = error;
	  console.log('ex---', ex);
	  throw ex;
	} catch (exi) {
	  console.log('exi---', exi);
	  throw exi;
	}
  }
}
